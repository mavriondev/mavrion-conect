export const auditLogs = pgTable("audit_logs", {
  id: serial("id").primaryKey(),
  orgId: integer("org_id").references(() => organizations.id),
  userId: integer("user_id").references(() => users.id),
  userName: text("user_name").notNull(),   // snapshot do nome no momento
  entity: text("entity").notNull(),         // "deal" | "lead" | "company" | "asset" | "connector"
  entityId: integer("entity_id").notNull(),
  entityTitle: text("entity_title"),        // snapshot do título/nome no momento
  action: text("action").notNull(),         // "created" | "updated" | "deleted" | "stage_changed" | "status_changed"
  changesJson: jsonb("changes_json").default({}), // { field: { from: X, to: Y } }
  createdAt: timestamp("created_at").defaultNow(),
});

export const insertAuditLogSchema = createInsertSchema(auditLogs).omit({ id: true, createdAt: true });
export type AuditLog = typeof auditLogs.$inferSelect;
export type InsertAuditLog = z.infer<typeof insertAuditLogSchema>;

**2. Criar helper `server/audit.ts`:**
```typescript
import { db } from "./db";
import { auditLogs } from "@shared/schema";

interface AuditOptions {
  orgId: number;
  userId: number;
  userName: string;
  entity: "deal" | "lead" | "company" | "asset" | "connector";
  entityId: number;
  entityTitle?: string;
  action: "created" | "updated" | "deleted" | "stage_changed" | "status_changed";
  changes?: Record<string, { from: any; to: any }>;
}

export async function audit(opts: AuditOptions) {
  try {
    await db.insert(auditLogs).values({
      orgId: opts.orgId,
      userId: opts.userId,
      userName: opts.userName,
      entity: opts.entity,
      entityId: opts.entityId,
      entityTitle: opts.entityTitle,
      action: opts.action,
      changesJson: opts.changes || {},
    });
  } catch (err) {
    // Audit nunca deve quebrar a operação principal
    console.error("[Audit] Falha ao registrar:", err);
  }
}

// Helper para calcular diff entre dois objetos
export function diff(before: Record<string, any>, after: Record<string, any>, fields: string[]) {
  const changes: Record<string, { from: any; to: any }> = {};
  for (const field of fields) {
    if (JSON.stringify(before[field]) !== JSON.stringify(after[field])) {
      changes[field] = { from: before[field], to: after[field] };
    }
  }
  return changes;
}
```

**6. Adicionar ao `IStorage` e `DatabaseStorage` em `server/storage.ts`:**
```typescript
// Interface:
getAuditLogs(orgId: number, filters?: { entity?: string; entityId?: number; limit?: number }): Promise<AuditLog[]>;

// Implementação:
async getAuditLogs(orgId: number, filters?: { entity?: string; entityId?: number; limit?: number }) {
  let query = db.select().from(auditLogs)
    .where(eq(auditLogs.orgId, orgId))
    .orderBy(desc(auditLogs.createdAt))
    .limit(filters?.limit || 100);
  
  // Aplicar filtros dinamicamente com Drizzle
  return query;
}
```

**7. Criar endpoint `GET /api/audit-logs` em `server/routes/system.ts`:**
```typescript
app.get("/api/audit-logs", async (req, res) => {
  if (!req.isAuthenticated()) return res.status(401).json({ message: "Unauthorized" });
  const user = req.user as any;
  const { entity, entityId, limit } = req.query;
  
  const logs = await storage.getAuditLogs(user.orgId, {
    entity: entity as string,
    entityId: entityId ? Number(entityId) : undefined,
    limit: limit ? Math.min(Number(limit), 500) : 100,
  });
  
  res.json(logs);
});
```
