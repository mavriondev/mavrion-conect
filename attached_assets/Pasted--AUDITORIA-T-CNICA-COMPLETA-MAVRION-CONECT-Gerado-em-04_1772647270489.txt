Se o endpoint não existir em nenhum arquivo, adicione-o ao arquivo
de rotas que contém `/api/sdr/queue` (provavelmente server/routes/index.ts
ou server/routes/leads.ts). Adicione:

```typescript
app.post("/api/sdr/leads/:id/promote", async (req, res) => {
  if (!req.isAuthenticated()) return res.status(401).json({ message: "Não autenticado" });
  try {
    const leadId = Number(req.params.id);
    const { title, pipelineType, stageId, amountEstimate, probability, source, description } = req.body;

    if (!title || !pipelineType || !stageId) {
      return res.status(400).json({ message: "title, pipelineType e stageId são obrigatórios" });
    }

    // Busca lead para pegar o companyId
    const leads = await db.select().from(leadsTable).where(eq(leadsTable.id, leadId));
    if (!leads[0]) return res.status(404).json({ message: "Lead não encontrado" });
    const lead = leads[0];

    // Cria o deal
    const deal = await storage.createDeal({
      orgId: lead.orgId!,
      pipelineType,
      stageId,
      title,
      amountEstimate: amountEstimate || null,
      probability: probability || 50,
      source: source || "SDR",
      description: description || null,
      companyId: lead.companyId || null,
    });

    // Marca lead como convertido
    await storage.updateLead(leadId, "converted");

    res.status(201).json(deal);
  } catch (err) {
    console.error("Promote lead error:", err);
    res.status(500).json({ message: "Erro ao promover lead" });
  }
});
```
