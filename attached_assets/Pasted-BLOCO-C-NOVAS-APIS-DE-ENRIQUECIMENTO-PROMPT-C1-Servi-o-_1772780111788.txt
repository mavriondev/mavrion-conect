/**
 * IBAMA — Embargos ambientais e autos de infração
 * API pública: https://dadosabertos.ibama.gov.br
 */

export interface IbamaEmbargoResult {
  cpfCnpj: string;
  embargos: Array<{
    numAuto: string;
    dataInfracao: string;
    municipio: string;
    uf: string;
    area: number;
    situacao: string;
    tipInfracao: string;
  }>;
  totalEmbargos: number;
  temEmbargo: boolean;
  fonte: string;
  consultadoEm: string;
}

export async function consultarEmbargoIbama(
  cpfCnpj: string
): Promise<IbamaEmbargoResult | null> {
  const doc = cpfCnpj.replace(/\D/g, "");
  try {
    const res = await fetch(
      `https://dadosabertos.ibama.gov.br/dados/SIFISC/auto_infracao/auto_infracao.json`,
      { signal: AbortSignal.timeout(10000) }
    );
    // API IBAMA retorna CSV/JSON público — filtrar pelo CPF/CNPJ
    // Fallback: consultar via SISCOM (sistema de controle ambiental)
    if (!res.ok) return null;
    const data = await res.json() as any;

    const registros = (data?.data || data || []).filter((r: any) =>
      (r.cpf_cnpj || r.CPF_CNPJ || "").replace(/\D/g, "") === doc
    );

    return {
      cpfCnpj: doc,
      embargos: registros.map((r: any) => ({
        numAuto: r.num_auto || r.NUM_AUTO || "",
        dataInfracao: r.dat_infracao || r.DAT_INFRACAO || "",
        municipio: r.municipio || r.MUNICIPIO || "",
        uf: r.uf || r.UF || "",
        area: Number(r.area_embargada || 0),
        situacao: r.des_situacao || r.DES_SITUACAO || "",
        tipInfracao: r.des_infracao || r.DES_INFRACAO || "",
      })),
      totalEmbargos: registros.length,
      temEmbargo: registros.length > 0,
      fonte: "IBAMA — Dados Abertos",
      consultadoEm: new Date().toISOString(),
    };
  } catch {
    return null;
  }
}

export async function consultarEmbargoIbamaCoordenadas(
  lat: number,
  lon: number
): Promise<{ temEmbargo: boolean; embargos: any[]; fonte: string } | null> {
  try {
    // Consulta via WFS IBAMA com bbox ao redor das coordenadas
    const delta = 0.01;
    const bbox = `${lon - delta},${lat - delta},${lon + delta},${lat + delta}`;
    const url = `https://geoserver.ibama.gov.br/geo/ows?service=WFS&version=1.0.0&request=GetFeature&typeName=embargos_ibama&outputFormat=application%2Fjson&bbox=${bbox},EPSG:4326`;

    const res = await fetch(url, { signal: AbortSignal.timeout(10000) });
    if (!res.ok) return null;
    const data = await res.json() as any;
    const features = data?.features || [];

    return {
      temEmbargo: features.length > 0,
      embargos: features.map((f: any) => f.properties),
      fonte: "IBAMA GeoServer",
    };
  } catch {
    return null;
  }
}
