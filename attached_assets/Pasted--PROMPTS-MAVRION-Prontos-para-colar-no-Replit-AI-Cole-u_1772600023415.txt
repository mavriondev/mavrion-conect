app.get("/api/prospeccao/reversa", async (req, res) => {
  if (!req.isAuthenticated()) return res.status(401).json({ message: "Não autenticado" });
  try {
    const { tipo, estado, substancia } = req.query as Record<string, string>;

    const CNAE_POR_TIPO: Record<string, number[]> = {
      MINA:            [710, 890, 810, 600],
      TERRA:           [111, 112, 113, 114, 115, 116, 119, 121, 151],

    params.set("limit", "20");
    params.set("head.eq", "true");
    params.set("status.id.in", "2");
    if (cnaesCodes.length > 0) params.set("mainActivity.id.in", cnaesCodes.join(","));
    if (estado) params.set("address.state.in", estado.toUpperCase());
    if (tipo === "MINA" && substancia) params.set("names.in", substancia);

    const response = await fetch(`https://api.cnpja.com/office?${params.toString()}`, { headers: cnpjaHeaders() });
    if (!response.ok) {
      const err = await response.json().catch(() => ({})) as any;
      return res.status(response.status).json({ message: err.message || "Erro na busca reversa" });
    }

    const data = await response.json() as any;
    const saved = await db.select({ cnpj: companies.cnpj }).from(companies);
    const savedSet = new Set(saved.map((c: any) => c.cnpj).filter(Boolean));

    const results = (data.records || []).map((r: any) => ({

      legalName: r.company?.name || "",
      tradeName: r.alias || null,
      porte: r.company?.size?.text || null,
