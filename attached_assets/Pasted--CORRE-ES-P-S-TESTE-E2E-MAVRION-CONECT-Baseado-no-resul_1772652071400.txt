Impacto: Kanban com filtro de pipeline retorna lista errada.

Localize o método getDeals e substitua por completo:

```typescript
async getDeals(pipelineType?: string): Promise<(Deal & { company: Company | null })[]> {
  const baseQuery = db
    .select({ deal: deals, company: companies })
    .from(deals)
    .leftJoin(companies, eq(deals.companyId, companies.id));

  const rows = pipelineType
    ? await baseQuery
        .where(eq(deals.pipelineType, pipelineType))
        .orderBy(desc(deals.createdAt))
    : await baseQuery
        .orderBy(desc(deals.createdAt));

  return rows.map(r => ({ ...r.deal, company: r.company }));
}
```

    const id = Number(req.params.id);
    // Verificar se empresa tem deals ou leads vinculados

      await audit({
        userId: user.id, userName: user.username,
        entity: "company", entityId: id, entityTitle: "Empresa removida",
        action: "deleted",
      });
    }
    res.status(204).end();
  } catch (err) {
    res.status(500).json({ message: "Erro ao remover empresa" });
